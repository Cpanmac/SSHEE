//小预告：在下一版本中（又可能会是layer1.9），将会在拓展模块集成layer.login
            
var loginhtml, validation = function(){
    //验证写在这
    $('.BB_button').on('click', function(){
        return false; //此处只作演示
    });
};
var page = {};
layer.login = function(options){
    options = options || {};
    $.layer({
        type: 1,
        title: '用户登录',
        offset: [($(window).height() - 290)/2+'px', ''],
        border : [5, 0.5, '#666'],
        area: ['450px','290px'],
        shadeClose: true,
        page: page
    });
};
$('#poplogin').on('click', function(){
    //如果已经请求过，则直接读取缓存节点
    if(loginhtml){
        page.html = loginhtml;
    } else {
        page.url = 'test/uplogin.html'
        page.ok = function(datas){
            loginhtml = datas; //保存登录节点
            validation();
        }
    }
    layer.login();    
});
            
var offset = [10, 10], layerLoop = function(){
    //获取随机宽高，仅为了区分层
    var area = [(100+Math.random()*800)+'px', (100+Math.random()*400)+'px'];
    
    //坐标叠加，仅为了区分层
    offset[0] += 15;
    offset[1] += 15;

    var index = $.layer({
        type:1,
        fix: false,
        area: area,
        moveOut: true,
        zIndex: layer.zIndex,
        shade: [0],
        title: '层：'+layer.index,
        offset: [offset[0]+'px', offset[1]+'px'],
        //以下对html做转义是为了防止被代码演示解析，实际应用时无需转义
        page: {html: '<\a href="javascript:;" id="next" class="next">点击继续弹出窗口\<\/a> <\a style="margin-top:10px;" class="btns" href="javascript:;" onclick="layer.closeAll();">关闭所有\<\/a>'}, 
        success: function(othis){
            var next = othis.find('#next'), tit = othis.find('.xubox_title');
            next.on('click', function(event){
                layerLoop();
            });
            
            //开启点击使当前窗口置顶功能
            layer.setTop(othis);
        }
    });
};
$('#looplayer').on('click', layerLoop);

//关闭所有层用此方法：layer.closeAll();
            
            
//此功能已集成到layer拓展模块，即layer.tab(options)，以下只是演示一个简单的tab弹层是如何制作而成的
var tabi;
$('#maketab').on('click', function(){
    tabi = $.layer({
        type:1,
        border: [0],
        area: ['602px', 'auto'],
        title: false,
        shade : [0.7 , '#000' , true],
        move: '.tabmove',
        closeBtn: false,
        page: {dom: '.tab'}
    });    
});

//切换事件
var btn = $('.tabtit').children(), main = $('.tab_main').children(), close = $('.tabclose');
btn.on('click', function(){
    var othis = $(this), index = othis.index();
    othis.addClass('tabnow').siblings().removeClass('tabnow');
    main.eq(index).show().siblings().hide();
});
//关闭层
close.on('click', function(){
    layer.close(tabi);
});
